  def compute_I_n(n, degeneracies=None):
      '''
      Computes the integral from the Dyson series.

      n: The order of the term
      degeneracies: A dictionary mapping equal energies E_mk, for example {E_m1: E_psi}

      Returns: A sympy expression for the evaluated integral
      '''

      if n < 1:
          raise ValueError("n must be at least 1")

      # Define the main symbols
      T = sp.Symbol('T', real=True, positive=True)
      s = sp.Symbol('s')
      E_psi = sp.Symbol('E_psi', real=True)

      # Define the list of intermediate energies E_{m_1} to E_{m_{n-1}}
      E_m = [sp.Symbol(f'E_m{k}', real=True) for k in range(1, n)]

      # Construct the Laplace transform of I_n
      # F(s) = 1 / ( (s+iE_psi)^2 \Prod_j(s+iE_{m_j}) )
      F_s = 1 / (s + sp.I*E_psi)**2
      for E_mk in E_m:
          F_s /= (s + sp.I*E_mk)

      # Apply known degeneracies
      if degeneracies:
          F_s = F_s.subs(degeneracies)

      # Identify all the poles
      poles = [-sp.I*E_psi, -sp.I*E_psi]
      for E_mk in E_m:
          pole = -sp.I*E_mk
          if degeneracies:
              pole = pole.subs(degeneracies)
          poles.append(pole)

      # Deduplicate poles
      unique_poles = list(set(poles))

      # Inverse Laplace transform
      integrand = F_s * sp.exp(s * T)
      I_n = 0
      for p in unique_poles:
          # m is the multiplicity of the pole
          m = poles.count(p)
          term = integrand * ((s-p)**m)

          # Take the (m-1)th derivative
          if m > 1:
              term = sp.diff(term, s, m-1)

          # Take the limit s --> p
          residue = (1 / sp.factorial(m-1)) * term.subs({s: p})

          I_n += residue

      return I_n

  def get_callable_I_n(n, expr):
      # Define the main symbols
      T = sp.Symbol('T', real=True, positive=True)
      E_psi = sp.Symbol('E_psi', real=True)

      # Define the list of intermediate energies E_{m_1} to E_{m_{n-1}}
      E_m = [sp.Symbol(f'E_m{k}', real=True) for k in range(1, n)]

      return sp.lambdify([T, E_psi, E_m], expr)

  def inc_characteristic_tuple(t, n):
      carry = True
      t = list(t)
      for i in range(n-1):
          ti = t[i]
          t[i] = ti ^ carry
          carry = ti and carry
          if not carry:
              break
      return tuple(t)

  def dict_union_from_ch_tuple(ch_tuple, basic_dict_list, n):
      dictionary = {}
      for i in range(n-1):
          if ch_tuple[i]:
              dictionary = dictionary | basic_dict_list[i]
      return dictionary

  def get_degeneracy_dicts(n):
      E_psi = sp.Symbol('E_psi', real=True)
      E_m = [sp.Symbol(f'E_m{k}', real=True) for k in range(1, n)]
      # Create a list of single-entry dictionaries.
      # Other dictionaries will be unions of these.
      basic_dict_list = [{E_mk: E_psi} for E_mk in E_m]
      ch_tuple = (False,) * (n-1)
      dict_list = [(ch_tuple, {})]
      while ch_tuple != (True,) * (n-1):
          ch_tuple = inc_characteristic_tuple(ch_tuple, n)
          dict_list.append((ch_tuple, dict_union_from_ch_tuple(ch_tuple, basic_dict_list, n)))
      return dict_list

  def hamiltonian_0(A1, A2, B):
      A1, A2, B = complex(A1), complex(A2), complex(B)
      M1 = A1.conjugate() * np.kron(sigma_m, creation)  +  A1 * np.kron(sigma_p, annihilation)
      M2 = A2.conjugate() * np.kron(sigma_m, creation @ creation)  +  A2 * np.kron(sigma_p, annihilation @ annihilation)
      Mb = B.conjugate() * np.kron(sigma_m, fock_ident) + B * np.kron(sigma_p, fock_ident)
      return M1 + M2 + Mb

  # There are ten Hermitian perturbations I want to consider, which come from nonzero chi^(3) or imperfect phase matching in the nonlinear optics implementation:
  # [ a^d a,     sigma_z,          a^d + a,       (a^d)^2 + a^2,  (a^d)^2 a + a^d a^2,
  #   (a^d a)^2, (a^d + a)sigma_z, a^d a sigma_z, spin_up,        a^d a spin_up ]
  hermitian_perturbation_ops = [
      np.kron(spin_ident, number),
      np.kron(sigma_z, fock_ident),
      np.kron(spin_ident, creation + annihilation),
      np.kron(spin_ident, creation @ creation + annihilation @ annihilation),
      np.kron(spin_ident, creation @ number + number @ annihilation),
      np.kron(spin_ident, number @ number),
      np.kron(sigma_z, creation + annihilation),
      np.kron(sigma_z, number),
      np.kron(spin_up, fock_ident),
      np.kron(spin_up, number)

  non_hermitian_perturbation_ops = [

  def normalised_wavefunc(wavefunc):
      norm = sum(wavefunc * wavefunc.conj())
      return wavefunc / np.sqrt(norm)

  def coherent_spin_down(alpha):
      wavefunc = np.zeros(2*fock_size, dtype=complex)
      wavefunc[fock_size] = 1
      for n in range(1, fock_size):
          wavefunc[fock_size + n] = wavefunc[fock_size + n - 1] * alpha / np.sqrt(n)
      return normalised_wavefunc(wavefunc)

* Perturbation theory

This code block will use the I_n implementation and solution to the unperturbed system to calculate the probability of survival for topological states. It works by creating a lambda for each term in the Dyson series that takes in the magnitudes of the perturbation and returns the numerical value of that term. Then the lambdas can be applied to a range of values.

#+begin_src python :session perturbation_theory :results output
  def I_n_for_set_of_energy_values(n, T, E_psi, idx_list, num_of_degeneracies):
      energies_list = [energies[idx] for idx in idx_list]
      #print(energies_list)
      if n not in callable_I_n_table:
          add_order_to_callable_I_n_table(n)
      return callable_I_n_table[n][num_of_degeneracies](T, E_psi, energies_list)

  def product_of_matrix_elements(perturbation_idx, eigenstate_idx_list, psi_idx):
      if len(eigenstate_idx_list) == 0:
          return hermitian_perturbation_matrices[perturbation_idx][psi_idx,psi_idx]
      product =\
          hermitian_perturbation_matrices[perturbation_idx][psi_idx,eigenstate_idx_list[0]] * \
          hermitian_perturbation_matrices[perturbation_idx][eigenstate_idx_list[-1],psi_idx]
      for i in range(len(eigenstate_idx_list) - 1):
          product *= hermitian_perturbation_matrices[perturbation_idx]\
              [eigenstate_idx_list[i],eigenstate_idx_list[i+1]]
      return product

  def inc_residual_eigenstate_idx_list(idx_list, append_idx):
      if len(idx_list) == 0:
          return None
      if idx_list[-1] == len(energies) - 1:
          return inc_residual_eigenstate_idx_list(idx_list[:-1], 0)
      idx_list[-1] += 1
      if idx_list[-1] == topological_state_indices[0]:
          idx_list[-1] += len(topological_state_indices)
      if append_idx != None:
          idx_list.append(append_idx)
      return idx_list

  def dyson_term(n, perturbation_idx, psi_idx, T, E_psi):
      dyson_sum = 0
      for num_of_degeneracies in range(n):
          degenerate_set = [psi_idx] * num_of_degeneracies
          residual_eigenstate_idx_list = [0] * (n - 1 - num_of_degeneracies)
          while residual_eigenstate_idx_list != None:
              idx_list = degenerate_set + residual_eigenstate_idx_list
              product = product_of_matrix_elements(perturbation_idx, idx_list, psi_idx)
              integral = I_n_for_set_of_energy_values(n, T, E_psi, idx_list, num_of_degeneracies)
              dyson_sum += product * integral
              residual_eigenstate_idx_list =\
                  inc_residual_eigenstate_idx_list(residual_eigenstate_idx_list, None)
      return (-1j)**n * dyson_sum
